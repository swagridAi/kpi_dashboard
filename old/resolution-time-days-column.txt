// ResolutionTimeDays - Calculated Column for Fact_Ticket_Summary
// Calculates the difference in days between ticket creation and resolution

ResolutionTimeDays = 
VAR CreatedDate = Fact_Ticket_Summary[created]
VAR ResolvedDate = Fact_Ticket_Summary[resolution_date]
VAR CurrentDate = NOW()

// Handle different scenarios for resolution time calculation
VAR ResolutionTimeDays = 
    SWITCH(
        TRUE(),
        
        // Case 1: Ticket is resolved - calculate actual resolution time
        ResolvedDate <> BLANK(),
            DATEDIFF(CreatedDate, ResolvedDate, DAY),
        
        // Case 2: Ticket not resolved - calculate current age
        ResolvedDate = BLANK(),
            DATEDIFF(CreatedDate, CurrentDate, DAY),
        
        // Default case (should not occur with proper data)
        BLANK()
    )

// Ensure non-negative values (handle any data quality issues)
VAR FinalResolutionDays = 
    IF(
        ResolutionTimeDays < 0,
        0,
        ResolutionTimeDays
    )

RETURN FinalResolutionDays
