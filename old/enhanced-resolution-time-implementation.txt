// Enhanced ResolutionTimeDays with Business Day Support
// Supports both calendar days and business days for comprehensive SLA analysis

// Primary calculated column (Calendar Days)
ResolutionTimeDays = 
VAR CreatedDate = Fact_Ticket_Summary[created]
VAR ResolvedDate = Fact_Ticket_Summary[resolution_date]
VAR CalculationResult = 
    DATEDIFF(
        CreatedDate,
        COALESCE(ResolvedDate, NOW()),
        DAY
    )
RETURN 
    MAX(CalculationResult, 0)  // Ensure non-negative values

// Additional calculated column for Business Days
ResolutionTimeBusinessDays = 
VAR CreatedDate = Fact_Ticket_Summary[created]
VAR ResolvedDate = COALESCE(Fact_Ticket_Summary[resolution_date], NOW())
VAR StartDate = DATE(YEAR(CreatedDate), MONTH(CreatedDate), DAY(CreatedDate))
VAR EndDate = DATE(YEAR(ResolvedDate), MONTH(ResolvedDate), DAY(ResolvedDate))

// Count business days between dates
VAR BusinessDaysCount = 
    SUMX(
        DATESBETWEEN(Dim_Date[Date], StartDate, EndDate),
        IF(Dim_Date[IsBusinessDay] = TRUE, 1, 0)
    )

RETURN BusinessDaysCount

// Calculated column for Resolution Status Categorization
ResolutionStatus = 
SWITCH(
    TRUE(),
    
    // Case 1: Resolved tickets
    Fact_Ticket_Summary[resolution_date] <> BLANK(),
        "Resolved",
    
    // Case 2: Open tickets within SLA
    AND(
        Fact_Ticket_Summary[resolution_date] = BLANK(),
        Fact_Ticket_Summary[ResolutionTimeDays] <= RELATED(Default_SLA_Table[SLA_Days])
    ),
        "Open - Within SLA",
    
    // Case 3: Open tickets approaching SLA (80% threshold)
    AND(
        Fact_Ticket_Summary[resolution_date] = BLANK(),
        Fact_Ticket_Summary[ResolutionTimeDays] > RELATED(Default_SLA_Table[SLA_Days]) * 0.8,
        Fact_Ticket_Summary[ResolutionTimeDays] <= RELATED(Default_SLA_Table[SLA_Days])
    ),
        "Open - At Risk",
    
    // Case 4: Open tickets breaching SLA
    AND(
        Fact_Ticket_Summary[resolution_date] = BLANK(),
        Fact_Ticket_Summary[ResolutionTimeDays] > RELATED(Default_SLA_Table[SLA_Days])
    ),
        "Open - SLA Breach",
    
    // Default case
    "Unknown"
)

// Calculated column for SLA Progress Percentage
SLA_Progress_Percentage = 
VAR SLATarget = RELATED(Default_SLA_Table[SLA_Days])
VAR ActualDays = Fact_Ticket_Summary[ResolutionTimeDays]
VAR ProgressPercentage = 
    IF(
        SLATarget > 0,
        MIN((ActualDays / SLATarget) * 100, 200),  // Cap at 200% for breached tickets
        100
    )
RETURN ProgressPercentage
